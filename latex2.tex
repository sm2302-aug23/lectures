\hypertarget{interlude-optional-arguments}{%
\subsection{Interlude: Optional
arguments}\label{interlude-optional-arguments}}

\begin{frame}[fragile]{Interlude: Optional arguments}
\begin{itemize}
\item
  We use square brackets \framebox{\texttt{[}} \framebox{\texttt{]}} for
  \emph{optional} arguments, instead of braces \framebox{\texttt{\{}}
  \framebox{\texttt{\}}}.
\item
  \VERB|\BuiltInTok{\textbackslash{}includegraphics}| accepts optional
  arguments that allow you to transform the image when it is included.
  For example,
  \VERB|\NormalTok{width=0.3}\FunctionTok{\textbackslash{}textwidth}|
  makes the image take up 30\% of the width of the surrounding text
  (\VERB|\FunctionTok{\textbackslash{}textwidth}|).
\item
  \VERB|\BuiltInTok{\textbackslash{}documentclass}| accepts optional
  arguments, too. E.g.
  \VERB|\BuiltInTok{\textbackslash{}documentclass}\NormalTok{[12pt,twocolumn]\{}\ExtensionTok{article}\NormalTok{\}}|
  makes the text bigger (12pt) and puts it into two columns.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Floats}
\protect\hypertarget{floats}{}
\begin{itemize}
\item
  Allow \LaTeX~to decide where the figure will go (it can ``float'').
\item
  You can also give the figure a caption, which can be referenced with
  \VERB|\KeywordTok{\textbackslash{}ref}|.
\item
  For more on floats, visit
  \href{https://www.overleaf.com/learn/latex/Positioning_images_and_tables}{this
  link}.
\end{itemize}

\subsection{Title, author, date}\label{title-author-date}}

\begin{frame}[fragile]{Title, author, date}
\begin{itemize}
\item
  Tell \LaTeX~the \VERB|\FunctionTok{\textbackslash{}title}| and
  \VERB|\FunctionTok{\textbackslash{}author}| names in the preamble.
  Note that author names are separated by
  \VERB|\FunctionTok{\textbackslash{}and}|.
\item
  The \texttt{\textbackslash{}date} command can be used to manually
  specify the date, or use
  \VERB|\FunctionTok{\textbackslash{}date}\NormalTok{\{}\FunctionTok{\textbackslash{}today}\NormalTok{\}}|
  for today's date.
\item
  Then use \VERB|\FunctionTok{\textbackslash{}maketitle}| (inserted just
  after
  \VERB|\KeywordTok{\textbackslash{}begin}\NormalTok{\{}\ExtensionTok{document}\NormalTok{\}}|)
  to actually create the title.
\end{itemize}

\hypertarget{the-bibtex-format}{%
\subsection{The bibtex format}\label{the-bibtex-format}}

\begin{frame}[fragile]{The bibtex format}
\begin{itemize}
\tightlist
\item
  \LaTeX~works off a `bibtex' database format:
\end{itemize}

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{@article\{lowry1951protein,}
\NormalTok{  title=\{Protein measurement with the Folin phenol reagent\},}
\NormalTok{  author=\{Lowry, OH and Rosebrough, NJ and Farr, AL and }
\NormalTok{          Randall, RJ\},}
\NormalTok{  journal=\{Journal of Biological Chemistry\},}
\NormalTok{  volume=\{193\},}
\NormalTok{  pages=\{265{-}{-}275\},}
\NormalTok{  year=\{1951\}}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\begin{itemize}
\item
  Each bibtex entry has a \emph{key} that you can use to reference it in
  the document. E.g., \texttt{lowry1951protein} is the key for the
  article above.
\item
  It's a good idea to use a key based on the name, year and title.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{A bib file}
\protect\hypertarget{a-bib-file}{}
\begin{itemize}
\item
  Collect all your references into a bib file, say \texttt{refs.bib}.
  This file should be in the folder together with your tex file.
\item
  Most reference managers (e.g.~Mendeley or Zotero) can export to bibtex
  format.
\item
  You can also use Google Scholar and do this manually.
\end{itemize}

\hypertarget{installing}{%
\subsection{\texorpdfstring{Installing
\LaTeX}{Installing }}\label{installing}}

\begin{frame}[fragile]{Installing \LaTeX}
\begin{itemize}
\item
  To run \LaTeX{} on your own computer, you'll want to use a \LaTeX{}
  \emph{distribution}. A distribution includes a \texttt{latex} program
  and (typically) several thousand packages.

  \begin{itemize}
  \tightlist
  \item
    On Windows: \href{http://miktex.org/}{Mik\TeX} or
    \href{http://tug.org/texlive/}{\TeX Live}
  \item
    On Linux: \href{http://tug.org/texlive/}{\TeX Live}
  \item
    On Mac: \href{http://tug.org/mactex/}{Mac\TeX}
  \end{itemize}
\item
  You'll also want a text editor with \LaTeX{} support. See
  \url{http://en.wikipedia.org/wiki/Comparison_of_TeX_editors} for a
  list of (many) options.
\item
  You'll also have to know more about how \texttt{latex} and its related
  tools work--see the resources on the next slide.
\end{itemize}
\end{frame}
